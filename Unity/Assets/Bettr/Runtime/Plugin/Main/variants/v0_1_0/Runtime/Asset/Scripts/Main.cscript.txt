require "math"

Main = {
    State = {}
}

function Main:new(o)
	o = o or {}
	setmetatable(o, self)
	self.__index = self
	return o
end

function Main:destroy()
	System.Print("Main destroy tile id={0}", self.id)
end

function Main:OnError(callStack)
	System.Error("Main Error CallStack: {0}", callStack)
end

function Main:Init()
    series
        do
            System.SetTargetFrameRate(30)
            RenderSettings.fog = false
            RenderSettings.skybox = null
            RenderSettings.sun = null
        end
    end
end

function Main:Login()
    series
        do
            BettrUserController.Login()
        end
    end
end

function Main:LoadMachine()
    series
        do
            System.Print("LoadMachine starting...")
            -- TODO: this should be from the user preferences
            local lobbyCard = BettrUser.LobbyCards[1]
            BettrAssetController.LoadScene(lobbyCard.MachineBundleName, lobbyCard.MachineBundleVariant, lobbyCard.MachineSceneName)
        end
    end
end

return Main:new()
